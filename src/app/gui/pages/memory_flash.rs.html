37/// Generuje odseparowany od GUI wątek do obsługi interakcji z programatorem (ST-Link).
38///
39/// Funkcja łączy się z pierwszym odnalezionym programatorem, ładuje plik .elf i asynchronicznie 
40/// raportuje postęp przesyłania paczek danych wprost do kanału MPSC stanu GUI.
41///
42/// # Parametry
43/// * `rx_command` - Kanał obierający komendy, np. żądanie startu z podaną ścieżką do pliku.
44/// * `tx_status`  - Kanał raportujący zmiany stanu, procenty ukończenia i błędy do aplikacji głównej.
45/// * `ctx`        - Referencja kontekstu egui, w celu przerysowania (repaint) ekranu po każdej zmianie stanu.
46pub fn spawn_flash_thread(
47    rx_command: mpsc::Receiver<FlashCommand>,
48    tx_status: mpsc::Sender<FlashStatus>,
49    ctx: egui::Context,
50) {
51    thread::spawn(move || {
52        while let Ok(FlashCommand::Start { path, chip }) = rx_command.recv() {
53            let _ = tx_status.send(FlashStatus::SearchProbe);
54            ctx.request_repaint();
55
56            let tx = tx_status.clone();
57            let ectx = ctx.clone();
58
59            let result = || -> Result<(), Box<dyn std::error::Error>> {
60                let lister = Lister::new();
61                let probes = lister.list_all();
62                let probe_info = probes.get(0).ok_or("Nie znaleziono programatora!")?;
63                
64                let mut probe = lister.open(probe_info)?;
65
66                probe.select_protocol(WireProtocol::Swd)?;
67                probe.set_speed(1000)?;
68
69                let mut session = probe.attach_under_reset(chip, probe_rs::Permissions::default())?;
70                
71                let total_bytes = AtomicU64::new(0);
72                let processed_bytes = AtomicU64::new(0);
73
74                let progress = FlashProgress::new(move |info| {
75                    use probe_rs::flashing::ProgressEvent;
76                    
77                    match info {
78                        ProgressEvent::StartedProgramming { length } => {
79                            total_bytes.store(length, Ordering::SeqCst);
80                            processed_bytes.store(0, Ordering::SeqCst);
81                        }
82                        ProgressEvent::PageProgrammed { size, .. } => {
83                            let current = processed_bytes.fetch_add(size as u64, Ordering::SeqCst) + size as u64;
84                            let total = total_bytes.load(Ordering::SeqCst);
85                            
86                            if total > 0 {
87                                let p = current as f32 / total as f32;
88                                let _ = tx.send(FlashStatus::Flashing(p));
89                                ectx.request_repaint();
90                            }
91                        }
92                        _ => {}
93                    }
94                });
95
96                let mut options = DownloadOptions::default();
97                options.progress = Some(progress);
98
99                probe_rs::flashing::download_file_with_options(
100                    &mut session, 
101                    &path, 
102                    probe_rs::flashing::Format::Elf, 
103                    options
104                )?;
105
106                let mut core = session.core(0)?;
107                core.reset()?;
108
109                Ok(())
110            }();
111
112            match result {
113                Ok(_) => { let _ = tx_status.send(FlashStatus::Finished); }
114                Err(e) => { 
115                    eprintln!("Flash error: {:?}", e); 
116                    let _ = tx_status.send(FlashStatus::Error(format!("{:?}", e))); 
117                }
118            }
119            ctx.request_repaint();
120        }
121    });
122}

124pub struct FlashPage {
125    tx_command: mpsc::Sender<FlashCommand>,
126    selected_file: Option<PathBuf>,
127    chip_name: String,
128}
129
130impl FlashPage {
131    /// Inicjalizuje nową stronę GUI do konfiguracji i nadzoru wgrywania programu na mikrokontroler.
132    pub fn new(ctx: &PageContext) -> Self {
133        Self {
134            tx_command: ctx.tx_flash.clone(),
135            selected_file: None,
136            chip_name: "STM32F429ZIT6".to_string(),
137        }
138    }
139}
140
141impl PageTrait for FlashPage {
142    fn update(&mut self, _ctx: &egui::Context, ui: &mut egui::Ui, state: &AppState) {
143        ui.heading("STM32 Firmware Uploader");
144
145        ui.horizontal(|ui| {
146            ui.label("Target Chip:");
147            ui.text_edit_singleline(&mut self.chip_name);
148        });
149
150        ui.horizontal(|ui| {
151            if ui.button("Pick ELF File").clicked() {
152                if let Some(path) = rfd::FileDialog::new()
153                    .add_filter("Executable", &["elf"])
154                    .pick_file() {
155                    self.selected_file = Some(path);
156                }
157            }
158            if let Some(path) = &self.selected_file {
159                ui.label(format!("{}", path.file_name().unwrap().to_string_lossy()));
160            }
161        });
162
163        ui.add_space(10.0);
164
165        let is_busy = matches!(state.flash_status, FlashStatus::SearchProbe | FlashStatus::Flashing(_));
166        let can_flash = self.selected_file.is_some() && !is_busy;
167
168        if ui.add_enabled(can_flash, egui::Button::new("Flash memory")).clicked() {
169            let _ = self.tx_command.send(FlashCommand::Start {
170                path: self.selected_file.clone().unwrap(),
171                chip: self.chip_name.clone(),
172            });
173        }
174
175        ui.add_space(10.0);
176
177        match &state.flash_status {
178            FlashStatus::SearchProbe => { 
179                ui.horizontal(|ui| {
180                    ui.spinner();
181                    ui.label("Searching for ST-Link...");
182                });
183            }
184            FlashStatus::Flashing(p) => { 
185                ui.add(egui::ProgressBar::new(*p).text(format!("Flashing: {:.0}%", p * 100.0))); 
186            }
187            FlashStatus::Finished => { 
188                ui.colored_label(Color32::GREEN, "Flash Successful!"); 
189            }
190            FlashStatus::Error(e) => { ui.colored_label(Color32::RED, format!("Error: {}", e)); }
191            _ => { ui.label("Ready"); }
192        }
193    }
194}
