24/// Struktura reprezentująca całą aplikację i przechowująca jej globalny stan.
25pub struct App {
26    current_page_type: PageType,
27    pages: HashMap<PageType, Box<dyn PageTrait>>,
28    state: AppState,
29    
30    rx_ser_stat: mpsc::Receiver<SerialStatus>,
31    rx_fl_stat: mpsc::Receiver<FlashStatus>,
32    rx_proc_stat: mpsc::Receiver<ProcedureStatus>,
33}
34
35impl App {
36    /// Tworzy nową instancję aplikacji.
37    ///
38    /// Konfiguruje współdzielony rejestr aktuatorów, uruchamia wątki odpowiedzialne za 
39    /// komunikację poprzez port szeregowy, programowanie mikrokontrolera oraz kontrolę procedur testowych, 
40    /// a także buduje wszystkie ekrany GUI z wykorzystaniem wzorca Factory.
41    pub fn new(config: AppConfig, ctx: egui::Context) -> Self {
42        let act_register = ActuatorsRegister::from_config(&config);
43
44        let (tx_ser_cmd, rx_ser_cmd) = mpsc::channel();
45        let (tx_ser_stat, rx_ser_stat) = mpsc::channel();
46        spawn_serial_thread(rx_ser_cmd, tx_ser_stat, act_register.clone(), ctx.clone());
47
48        let (tx_fl_cmd, rx_fl_cmd) = mpsc::channel();
49        let (tx_fl_stat, rx_fl_stat) = mpsc::channel();
50        spawn_flash_thread(rx_fl_cmd, tx_fl_stat, ctx.clone());
51
52        let (tx_proc_cmd, rx_proc_cmd) = mpsc::channel();
53        let (tx_proc_stat, rx_proc_stat) = mpsc::channel();
54        spawn_proc_thread(rx_proc_cmd, tx_proc_stat, tx_ser_cmd.clone(), config.clone(), ctx.clone(), act_register.clone());
55
56        let state = AppState {
57            serial_status: SerialStatus::Disconnected,
58            flash_status: FlashStatus::Idle,
59            proc_state: ProcedureStatus::Idle,
60            available_ports: vec![],
61        };
62
63        let page_ctx = PageContext {
64            config: config.clone(),
65            act_register,
66            tx_serial: tx_ser_cmd,
67            tx_flash: tx_fl_cmd,
68            tx_proc: tx_proc_cmd,
69        };
70
71        let act_page = PageFactory::create(PageType::Actuators, &page_ctx);
72        let serial_page = PageFactory::create(PageType::SerialPort, &page_ctx);
73        let flash_page = PageFactory::create(PageType::Flash, &page_ctx);
74        let proc_page = PageFactory::create(PageType::Procedures, &page_ctx);
75
76        let mut pages: HashMap<PageType, Box<dyn PageTrait>> = HashMap::new();
77        pages.insert(PageType::Actuators, act_page);
78        pages.insert(PageType::SerialPort,  serial_page);
79        pages.insert(PageType::Flash, flash_page);
80        pages.insert(PageType::Procedures, proc_page);
81
82        Self {
83            current_page_type: PageType::Actuators,
84            pages,
85            state,
86            rx_ser_stat,
87            rx_fl_stat,
88            rx_proc_stat,
89        }
90    }
91}
92
93impl eframe::App for App {
94    /// Główna pętla renderująca GUI.
95    ///
96    /// Aktualizuje stany pochodzące z zewnętrznych wątków i renderuje odpowiedni widok
97    /// bocznego panelu nawigacyjnego oraz centralnego panelu z wybraną zakładką.
98    fn update(&mut self, ctx: &egui::Context, _frame: &mut eframe::Frame) {
99        while let Ok(msg) = self.rx_ser_stat.try_recv() {
100            match msg {
101                SerialStatus::PortList(p) => self.state.available_ports = p,
102                status => self.state.serial_status = status,
103            }
104        }
105        while let Ok(status) = self.rx_fl_stat.try_recv() { self.state.flash_status = status; }
106        while let Ok(status) = self.rx_proc_stat.try_recv() { self.state.proc_state = status; }
107
108        egui::SidePanel::left("menu").show(ctx, |ui| {
109            if ui.selectable_label(self.current_page_type == PageType::Actuators, "Actuators").clicked() {
110                self.current_page_type = PageType::Actuators;
111            }
112            if ui.selectable_label(self.current_page_type == PageType::Procedures, "Procedures").clicked() {
113                self.current_page_type = PageType::Procedures;
114            }
115            if ui.selectable_label(self.current_page_type == PageType::SerialPort, "Serial Port").clicked() {
116                self.current_page_type = PageType::SerialPort;
117            }
118            if ui.selectable_label(self.current_page_type == PageType::Flash, "Flash").clicked() {
119                self.current_page_type = PageType::Flash;
120            }
121        });
122
123        egui::CentralPanel::default().show(ctx, |ui| {
124            if let Some(page) = self.pages.get_mut(&self.current_page_type) {
125                page.update(ctx, ui, &self.state);
126            }
127        });
128    }
